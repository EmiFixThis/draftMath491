	name={Unimodular Lattice},
	description={An integral lattice which has determinant value 1 or (-1).}
}

\newglossaryentry{unimodular lattice equivalency}{
	name={Unimodular Lattice Equivalency},
	description={In n-dimensional Euclidean space, the value of the determinant requires the fundamental domain of the lattice has a volume equal to one.}
}

\newglossaryentry{trapdoor function}{
	name={Trapdoor Function},
	description={The usual term trapdoor function is rather misleading. The trapdoor of a function is an added structure which provides information enabling the inverse of the one-way function to be easily computed.}
}

\newglossaryentry{one-way function}{
	name={One-way function},
	description={A one-way function is one in which it is easy to compute the output, but reversing the function to retrieve the input; i.e. taking the inverse should be hard to compute.}
}

\newglossaryentry{one-way function assumption}{
	name={One-way Function Assumption},
	description={The usual practice when constructing cryptographic schemes is to assume that a one-way function exists, and that we can construct a corresponding trapdoor function which provides some added information allowing the one-wayness of this function to be easily computed. However, in reality this assumption may indirectly depend on the unanswered `Does \ensuremath{$\mathbf{FP \stackrel{?}{=} FNP}$}', which would imply that \ensuremath{$\mathbf{P \stackrel{?}{=} NP}$}. However, it remains unknown if a result for \ensuremath{$\mathbf{P \stackrel{?}{=} NP}$} would imply the existence of one-way functions.}
}

    name={GapSVP-$\beta$ 1},
    description={$GapSVP_{\beta}$ is a \emph{promise problem} which differentiates between the instances of SVP where the answer is at most 1 or larger than $\beta$, where $\beta$ is allowed to be a fixed function of $f(n)$, such that $n$ is the number of vectors.}
}

\newglossaryentry{gapsvpb decisional}{
    name={GapSVP-$\beta$ 2},
    description={Given a basis $b$ for a lattice $\Lambda$, the $GapSVP-{\beta}$ decides if $\Lambda(b) \leq 1$ or $\Lambda{b} > \beta$, in all other cases the algorithm returns an error.}
}

\newglossaryentry{gapcvpb}{
    name={$GapCVP_{\beta}$},
    description={The $GapCVP_{\beta}$ algorithm takes a basis $b$ for a lattice $\Lambda$ and a vector $\mathbf{v}$ and answers whether or not there exists a lattice vector with a distance between itself and the given vector is at most one, or if every lattice vector has a distance greater than $\beta$ from the given vector.}
}

\newglossaryentry{ideal order theory}{
    name={Ideal Order Theoretic Notion},
    description={Given a non-empty subset S of a poset $(P, \leq)$, S is an ideal $I$, if for all $x \in I, y \leq x \Rightarrow y \in I$ (meaning, $I$ is a lower set). For all $x,y \in I$, there exists an element $z \in I$, such that $x \leq z$ and $\y \leq z$, (meaning, $I$ is a directed set).}
}

\newglossaryentry{ideal}{
    name={Ideal Order Theoretic Notion},
    description={In general for arbitrary posets defined over lattices, a subset $I \nequal \emptyset$ of a poset $(P, \leq)$ is an ideal if and only if $I$ is a lower set which is closed under finite joins (suprema), and if for all $x,y \in I$ the join $x \vee y$ is also in $I$.}
}

\newglossaryentry{lattice order theory}{
    name={Lattice (Order Theoretic Notion)},
    description={In order theory, a lattice is a partially ordered set (poset) in which the greatest lower bound called the join infimum and least upper bound (called the meet or supremum) both exist and are unique. It follows by completeness over lattices that every finite subset (non-empty) also contains both a join and a meet.}
}

\newglossaryentry{noncomputability}{
    name={noncomputability},
    description={Noncomputability refers to a type of problem for which a solution can be described but not computed. One example of a non-computable problem is the Halting Problem.}
}

\newglossaryentry{poset}{
	name={poset},
	description={A poset is a partial ordering of a set $S$ by a binary relation $\leq$, which is reflexive, antisymmetric, and transitive. That is, for two elements $a,b \in S$ the following hold: $a \leq a$ (reflexivity). If $a \leq b$ and $b \leq a$, then $a=b$ (anti-symmetry). If $a \leq b$ and $b \leq c$ then $a \leq c$ (transitivity).}
}

\newglossaryentry{asymmetric-key}{
    name={asymmetric key encryption},
    description={Asymmetric key encryption (public-key encryption) is a method of encryption which uses a pair of keys in order encrypt and decrypt messages. The public key may be used by anyone to encrypt messages for a certain user. The secret key, is known only to that user and allows the decryption of messages which were encrypted using the public key.}
}

\newglossaryentry{symmetric-key}{
    name={symmetric-key encryption},
    description={Symmetric key encryption involves the use of only one key which all communicating parties share. The key is secret, and performs both the encryption and decryption of messages. }
}

\newglossaryentry{shared secret}{
    name={shared secret},
    description={A shared secret is some secret piece of information which allows two or more parties to engage in secure communication.}
}

\newglossaryentry{secure channel}{

\newglossaryentry{authentic channel}{

    description={An authentic channel enables communication which may be resistant to alteration but not necessarily resistant to interception.}

\newglossaryentry{message space}{

\newglossaryentry{insecure channel}{

}
